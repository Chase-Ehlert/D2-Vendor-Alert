import mongoose from 'mongoose'

mongoose.set('strictQuery', false)

const UserSchema = new mongoose.Schema({
    bungie_username: String,
    discord_id: String,
    discord_channel_id: String,
    bungie_membership_id: String,
    destiny_id: String,
    destiny_character_id: String,
    refresh_expiration: String,
    refresh_token: String
})

/**
 * 
 * @typedef {Object} User
 * @property {string} bungie_username
 * @property {string} discord_id
 * @property {string} discord_channel_id
 * @property {string} bungie_membership_id
 * @property {string} destiny_id
 * @property {string} destiny_character_id
 * @property {string} refresh_expiration
 * @property {string} refresh_token
*/

export const User = mongoose.model("users", UserSchema)
